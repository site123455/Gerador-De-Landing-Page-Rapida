<script>
const canvas = document.getElementById("canvas");
const trash = document.getElementById("trash");

/* ===== SISTEMA DE PÃGINAS ===== */
let pages = [];
let currentPage = 0;

function addPage(){
  pages.push({elements:[], background:""});
  currentPage = pages.length-1;
  updatePageList();
  renderPage();
}

function updatePageList(){
  const select = document.getElementById("pageSelect");
  select.innerHTML="";
  pages.forEach((p,i)=>{
    const opt=document.createElement("option");
    opt.value=i;
    opt.text="PÃ¡gina "+(i+1);
    select.appendChild(opt);
  });
  select.value=currentPage;
}

function changePage(){
  currentPage = Number(pageSelect.value);
  renderPage();
}

function renderPage(){
  canvas.querySelectorAll(".draggable").forEach(e=>e.remove());
  canvas.style.backgroundImage = pages[currentPage].background;

  pages[currentPage].elements.forEach(el=>{
    canvas.appendChild(el);
    drag(el);
  });
}

/* ===== ARRASTAR + APAGAR ===== */
function drag(el){
let ox,oy,move=false;

el.onmousedown=e=>{
move=true;
ox=e.offsetX;
oy=e.offsetY;
};

document.onmousemove=e=>{
if(!move)return;

const r=canvas.getBoundingClientRect();
el.style.left=(e.clientX-r.left-ox)+"px";
el.style.top=(e.clientY-r.top-oy)+"px";

/* verifica colisÃ£o com lixeira */
const elRect = el.getBoundingClientRect();
const trashRect = trash.getBoundingClientRect();

const colidiu =
!(elRect.right < trashRect.left ||
  elRect.left > trashRect.right ||
  elRect.bottom < trashRect.top ||
  elRect.top > trashRect.bottom);

trash.classList.toggle("hover", colidiu);
};

document.onmouseup=()=>{
if(!move)return;
move=false;

/* se soltou na lixeira â†’ deletar */
const elRect = el.getBoundingClientRect();
const trashRect = trash.getBoundingClientRect();

const colidiu =
!(elRect.right < trashRect.left ||
  elRect.left > trashRect.right ||
  elRect.bottom < trashRect.top ||
  elRect.top > trashRect.bottom);

if(colidiu){
  pages[currentPage].elements =
    pages[currentPage].elements.filter(e=>e!==el);
  el.remove();
}

trash.classList.remove("hover");
};
}

/* ===== ELEMENTOS ===== */

}

/* ===== EXPORTAR SITE ===== */
function exportPages(){

pages.forEach((page,i)=>{

let html="";
page.elements.forEach(el=>{

if(el.classList.contains("text")){
html+=`<div style="position:absolute;left:${el.style.left};top:${el.style.top};font-size:28px;font-weight:600;">
${el.innerText}</div>`;
}

if(el.classList.contains("button")){
html+=`<a href="${el.dataset.link}" style="
position:absolute;
left:${el.style.left};
top:${el.style.top};
width:${el.offsetWidth}px;
height:${el.offsetHeight}px;
display:flex;align-items:center;justify-content:center;
background:#22c55e;
color:white;border-radius:8px;text-decoration:none;">
${el.innerText}</a>`;
}

if(el.classList.contains("image")){
const img=el.querySelector("img");
html+=`<img src="${img.src}" style="
position:absolute;left:${el.style.left};top:${el.style.top};
width:${el.offsetWidth}px;height:${el.offsetHeight}px;">`;
}

if(el.classList.contains("social")){
const img=el.querySelector("img");
html+=`<a href="${el.dataset.link}" style="
position:absolute;left:${el.style.left};top:${el.style.top};
width:${el.offsetWidth}px;height:${el.offsetHeight}px;border-radius:50%;overflow:hidden;">
<img src="${img.src}" style="width:100%;height:100%;">
</a>`;
}
});

const final=`

${html}

const a=document.createElement("a");
a.href=URL.createObjectURL(blob);
a.download="pagina"+(i+1)+".html";
a.click();

});

alert("Site exportado!");
}
